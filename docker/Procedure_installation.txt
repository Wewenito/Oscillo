#Start by creating the mysql container using the command below.
#To change the username, the db name, the password or ports, change the 'startMysqlDB.sh' file.

> bash startMysqlDB.sh

#Create the image for the container hosting the oscillo-app.
#This command needs to be done at the same level as the Dockerfile (may take a few minutes).

> docker build -t oscillo-app .

#Now we start the container & link it to the db container

> bash runOscillo.sh

#The app should now be deployed and ready to use on any web browser at :
#http://localhost:8000/oscillo
#http://127.0.0.1:8000/oscillo
